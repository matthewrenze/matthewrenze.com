    data = as.numeric(values),

    ncol = 28)

    x = 1:28, 
    y = 1:28, 
    z = first_image[,28:1], 
    col = gray((0:255) / 255))

4. Create a Fully-connected Deep Neural Network

(input) -> (128-relu) -> (64-relu) -> (10-softmax)

  1. Load the MXNet package
library(mxnet)
  1. Create the input layer
data <- mx.symbol.Variable("data")
  1. Create the 1st fully-connected hidden layer
fc1 <- mx.symbol.FullyConnected(
  data = data, 
  name = "fc1", 
  num_hidden = 128)
  1. Create the 1st activation function (relu)
act1 <- mx.symbol.Activation(
  data = fc1, 
  name = "relu1", 
  act_type = "relu")
  1. Create the 2nd fully-connected hidden layer
fc2 <- mx.symbol.FullyConnected(
  data = act1, 
  name = "fc2", 
  num_hidden = 64)
  1. Create the 2nd activation function (relu)
act2 <- mx.symbol.Activation(
  data = fc2, 
  name = "relu2", 
  act_type = "relu")
  1. Create the output layer
fc3 <- mx.symbol.FullyConnected(
  data = act2, 
  name = "fc3",
  num_hidden = 10)
  1. Create the output activation function (softmax)
softmax <- mx.symbol.SoftmaxOutput(
  data = fc3, 
  name = "sm")
  1. Set the MXNet random seed
mx.set.seed(0)
  1. Train the model
    Note: This could take several minutes
model <- mx.model.FeedForward.create(
  symbol = softmax, 
  X = train.X, 
  y = train.y,
  ctx = mx.cpu(), 
  num.round = 10, 
  array.batch.size = 100,
  array.layout = "colmajor",
  learning.rate = 0.07, 
  momentum = 0.9,  
  eval.metric = mx.metric.accuracy,
  initializer = mx.init.uniform(0.07),
  epoch.end.callback = mx.callback.log.train.metric(100))
